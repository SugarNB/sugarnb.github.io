```bash
#用户配置
#显示当前git配置
git config --list
git config -l

#设置全局用户名和邮箱
git config --global user.name "username"
git config --global user.email "xxx@xxx.com"

#查看特定配置
git config user.name
git config user.email

#仓库操作
1.创建与克隆
git init
git clone [url]
git clone [url] [dirname]

2.远程仓库管理
#查看远程仓库
git remote
git remote -v

#添加远程仓库
git remote add [别名] [url]

#修改远程仓库url
git remote set-url origin [new_url]

#删除远程仓库
git remote rm [repository]


#基本操作
1.文件操作
git add [file1] [file2] ...
git add [dir]
git add .

2.提交操作
git commit -m "提交描述"
#提交特定文件
git commit [file1][file2] -m "描述"

#分支与合并
1.分支管理
#查看分支
git branch #本地
git branch -r #远程
git branch -a#所有分支
git branch -a -v #显示详细信息

#创建分支
git branch [branch-name]

#切换分支
git checkout [branch-name]

#创建并切换分支
git checkout -b [branch-name]

#删除分支
git branch -d [branch-name] #安全删除
git branch -D [branch-name] #强制

#删除远程分支
git push origin --delete [branch-name]

2.合并与变基
#合并指定分支到当前分支
git merge [branch]
...


#推送与拉取
#拉取远程更新
git fetch [remote] #下载所有变动

#拉取并合并
git pull #默认拉取当前分支
git pull origin develop #从指定分支拉取

#推送本地分支
git push [remote] [branch]
git push -u origin --all #推送所有分支

#强制推送
git push [remote] --force

#推送标签
git push [remote] --tags


#状态与日志查看
#查看文件状态
git status
#查看简明状态
git status -s

#查看提交历史
git log
git log --oneline
git log --graph --oneline --all

#查看文件修改
git log -p [file]

#查看特定文件历史
git blame [file]

#查看引用日志
git reflog


#回滚和撤销
1.重置命令
#回滚到指定节点（不保留修改）
git reset --hard [commit-hash]

#回滚到指定节点（保留修改）
git reset --soft [commit-hash]

#回滚到上一个提交
git reset --hard HEAD^

#撤销暂存区文件
git reset HEAD [file]

#撤销工作区修改
git checkout --[file]

2.撤销提交
#创建撤销提交（推荐）
git revert [commit-hash]

#恢复暂存更改
git stash pop

#删除暂存
git stash drop


......

```
